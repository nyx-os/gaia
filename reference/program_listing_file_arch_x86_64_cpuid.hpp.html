#define CPUID_LEAF_EXTENDED_FEATURES 0x80000001
#define CPUID_PROC_EXTENDED_STATE_ENUMERATION 13

struct Cpuid {
  uint32_t eax;
  uint32_t ebx;
  uint32_t ecx;
  uint32_t edx;

  static inline Result<Cpuid, Error> cpuid(uint32_t leaf = 0,
                                           uint32_t subleaf = 0) {
    uint32_t cpuid_max;

    Cpuid ret;

    asm volatile("cpuid"
                 : "=a"(cpuid_max)
                 : "a"(leaf & 0x80000000)
                 : "rbx", "rcx", "rdx");

    if (leaf > cpuid_max) {
      return Err(Error::INVALID_PARAMETERS);
    }

    asm volatile("cpuid"
                 : "=a"(ret.eax), "=b"(ret.ebx), "=c"(ret.ecx), "=d"(ret.edx)
                 : "a"(leaf), "c"(subleaf));

    return Ok(ret);
  }
